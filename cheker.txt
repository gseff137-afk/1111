pause

# Функция для анимированной точки
function Show-Dots {
    param([int]$Seconds)
    $end = (Get-Date).AddSeconds($Seconds)
    while ((Get-Date) -lt $end) {
        Write-Host "." -NoNewline -Foreground Yellow
        Start-Sleep 1
    }
    Write-Host ""
}
